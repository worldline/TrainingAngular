Main Concepts About Templates - Property Binding - Declarations

  • An Angular app is a tree of Web Components
  • To be portable, a component has to be prameterizable
  • This is done by using the Property Binding mechanism
  • You can use the inputs attribute in the @Component annotation to declare a parameter for a component
import {Component} from '@angular/core'; import {BrowserModule} from '@angular/platform-browser'; import {SubCmp} from './sub-cmp'; @Component({ selector: 'app', template: '<sub-cmp name="john" ></sub-cmp>' }) class App {} export const ModuleDefs= { bootstrap: [App], declarations:[ App, SubCmp], imports: [BrowserModule] } import {Component} from '@angular/core'; @Component({ selector: 'sub-cmp', template: ` <h1> Name : {{name}}</h1> `, inputs: ['name'] }) export class SubCmp { name: string; }
