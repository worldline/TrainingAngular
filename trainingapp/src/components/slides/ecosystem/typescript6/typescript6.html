Ecosystem Typescript - Interfaces

Interface are a great feature to help design Oriented Objects applications. Moreover, their declaration is really flexible in typescript. You can give them a name or make anonymous one..;

let updateUserData= function(user:User, userData:{age:number, isMarried: boolean} ): void{ //... } updateUserData(new User('a', 'b'), {age:15, isMarried: true}) // OK interface UserData { age: number; isMarried: boolean; } let updateUserData= function(user: User, userData:UserData): void { //... } updateUserData(new User('a', 'b'), {age:15, isMarried: true}) // OK

You can also use interfaces to make functions, arrays and classes respect a contract (a class can implement several interfaces)

interface UpdateUserCallback{ function(updatedUser: User): void; } let myCallback: UpdateUserCallback= function(user: User): void { //do something }; //Standard arrays interface PersonArray{ [index: number]: Person; } let friends: PersonArray= [new Person('john', 'doe'), new Person('bob', 'sponge')]; let bob: Person = PersonArray[0]; //Indexed objects interface FriendsByName{ [index: string]: Person; }; let allFriends: FriendsByName= { 'john': new Person('john', 'doe'); 'bob': new Person('bob', 'sponge'); } interface PersonItf{ age: number; getFriends():PersonArray; } class PersonImpl1 implements PersonItf{ constructor(public age, public friends: PersonArray){} getFriends(): PersonArray= { return this.friends; } } //Array functions also work class PersonImpl1 implements PersonItf{ constructor(public age, public friends: PersonArray){} getFriends: (): PersonArray => this.friends; }
