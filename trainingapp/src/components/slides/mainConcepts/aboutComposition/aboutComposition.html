Main Concepts About Composition

  • Remember our Web Compoments composition purpose with the content tag?
  • Angular allows to do the same with the ng-content attribute
  • You can even get references on the content with @ContentChild and @ContentChildren
  • Note that you can also use @ViewChild and @ViewChildren to get access to your template children (no composition, just the child components of your current components)
import {Component, ContentChildren, AfterViewInit} from '@angular/core'; import {BrowserModule} from '@angular/platform-browser'; @Component({ selector: 'person', template: `hello {{name}}`, inputs: ['name'], }) class Person { name: string; } @Component({ selector: 'display-persons', template: ` <ng-content></ng-content> <p> We are displaying {{persons?.length}} persons </p> ` }) class DisplayPersons implements AfterViewInit{ @ContentChildren(Person) persons: QueryList<Person>; } @Component({ selector: 'app', template: ` <display-persons> <ul> <li><person [name]="'john'"></person></li> <li><person [name]="'bob'"></person></li> </ul> </display-persons> ` }) export class App {} export const ModuleDefs= { bootstrap: [App], imports: [BrowserModule], declarations: [App, DisplayPersons, Person] }
